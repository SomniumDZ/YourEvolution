import javafx.scene.image.Image; 
import javafx.scene.image.ImageView; 
 
  class entities.bacteria.BMod extends ImageView{
//     Классовая система модификаций 
    final String name; 
    float x, y; 
    int price; 
    boolean isOutside; 
 
    static entities.bacteria.BMod None = new entities.bacteria.BMod("None", new Image("res/BMods/None.png"),0,0,0,0, false);
    static entities.bacteria.BMod Flagellum = new entities.bacteria.BMod("Flagellum", new Image("res/BMods/BMod_Flagellum.png"),0,-199,-3,.25, true);
    static entities.bacteria.BMod Nucleus = new entities.bacteria.BMod("Nucleus", new Image("res/BMods/Nucleus-1.png"),40,0,0,1,false);
 
    private entities.bacteria.BMod(String name, Image iv, int price, float x, float y, double scale, boolean isOutside) {
        this.name = name; 
        this.price = price; 
        setImage(iv); 
        setTranslateX(x); 
        setTranslateY(y); 
        setScaleX(scale); 
        setScaleY(scale); 
        this.isOutside = isOutside; 
    } 
}
