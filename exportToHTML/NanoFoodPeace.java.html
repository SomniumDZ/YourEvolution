import javafx.geometry.Point2D; 
import javafx.scene.layout.Pane; 
import javafx.scene.paint.Color; 
import javafx.scene.shape.Rectangle; 
 
class food.NanoFoodPiece extends Pane {
 
    Point2D center_g; 
 
    food.NanoFoodPiece() {
        Rectangle graphics = new Rectangle(11, 11, Color.rgb(150, 144, 107)); 
        double scale = Math.random()+.5d; 
        graphics.setScaleX(scale); 
        graphics.setScaleY(scale); 
        getChildren().add(graphics); 
        setTranslateX(Math.random()*600); 
        setTranslateY(Math.random()*600); 
        center_g = new Point2D(getTranslateX()+graphics.getWidth(),getTranslateY()+graphics.getHeight()); 
    } 
 
    void destroy(){ 
        setTranslateX(700); 
    } 
}
