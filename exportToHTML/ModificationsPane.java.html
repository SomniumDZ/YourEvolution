import javafx.geometry.Pos; 
import javafx.scene.control.Button; 
import javafx.scene.layout.VBox; 
 
import java.util.HashMap; 
import java.util.Map; 
 
class controllers.ModificationsPane extends VBox {
    private Map<String, Button> map = new HashMap<>(); 
    controllers.ModificationsPane(){
        setSpacing(5); 
        setAlignment(Pos.CENTER); 
 
        for (entities.bacteria.BacteriaModifications next:entities.bacteria.BacteriaModifications.values()){
            map.put(next.name(), new Button(next.name())); 
        } 
        map.forEach((name, tag)->{ 
            getChildren().add(tag); 
            tag.setOnAction(event -> { 
                BTarget.addModification(entities.bacteria.BacteriaModifications.valueOf(name).mod);
                setVisible(false); 
            }); 
        }); 
    } 
    void reinitialize(){ 
        map.forEach((next,btm)->{ 
            boolean c = BTarget.satiety<entities.bacteria.BacteriaModifications.valueOf(next).mod.price;
                if (BTarget.modifications.contains(entities.bacteria.BacteriaModifications.valueOf(next).mod)||c){
                    getChildren().remove(btm); 
 
                }else if (!getChildren().contains(btm)) getChildren().add(btm); 
        }); 
    } 
}
